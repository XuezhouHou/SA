\newcommand{\bild}[4]{%
	\begin{figure}[#1]
		\centering
		\includegraphics
		[
		width=#2\textwidth,
		]
		{figures/#3}
		
		\caption{#4}
		\label{fig:#3}
	\end{figure}
}
%
%
\newcommand{\bildsvg}[4]{
	\begin{figure}[#1]
		\centering%
		\def\svgwidth{#2\columnwidth}%
		\input{figures/#3}%
		%
		\caption{#4}%
		\label{fig:#3}%
	\end{figure}	
}
